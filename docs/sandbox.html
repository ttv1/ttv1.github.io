     mu += delta/n
     m2 += delta*(x - mu)
     lo=min(lo,x)

     delta=(hi - lo)*epsilon
     if n < pause:

def novel(seq,epsilon=1.1):
  _,v1,_ = next(seq)
  while True:
    n,v2,sd = next(seq)
    if (v2 - v1)/(v1+1e-32) > epsilon:
      yield v2,n,sd
      v1 = v2
